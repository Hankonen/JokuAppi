

GIT!!!
git --version			// git versio
git config --list 		//näyttää tietoja
git config --global user.name "Nimesi tänne"
git config --global user.email "sposti tänne"	// kuka teki mitä, nämä tiedot tallentuvat committii?
git init			// luodaan paikallinen repo
ls -a				// näyttää piilotetun .git repon

git status			// näyttää tilanteen 
git add <tiedosto> <tiedosto2>	// lisää tiedostoja committiin
git add .			// lisää kaikki muuttuneet tiedostot committiin
git commit -m "viesti tähän"	// tekee uuden commitin
git log				// näyttää tehdyt commitit
git branch 			// tsekkaa branchit
git branch -b uusibranch 	// luo uuden branchin
git checkout <versionumero> 	// hyppää tiettyyn versioon (versionumero löytyy logista)
git checkout <branchnimi> 	// vaihtaa branchia
git merge <branchnimi> 		// yhdistää sivu branchin uudeksi pääbrachiks
git branch -d <branch-name> 	// poistaa branchin

git remote add origin https://github.com/t9tike/testi.git	// originin voi vaihtaa muuksi, lisää määränpää osoite pushiksi (luo ensin uusi repo githubissa josta saa osoitteen)
git remote -v 			// tsekkaa että yhteys on githubiin			
git push -u origin <branch nimi> //puskee shaibat githubii
git pull origin <branch nimi> 	// hakee shaibat githubista
